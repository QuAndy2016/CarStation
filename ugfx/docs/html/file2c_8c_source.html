17 static char *filenameof(char *fname) {
18  char *p;
19 
20 #ifdef WIN32
21  if (fname[1] == ':')
22  fname = fname+2;
23  p = strrchr(fname, '\\');
24  if (p) fname = p+1;
25 #endif
26  p = strrchr(fname, '/');
27  if (p) fname = p+1;
28  p = strchr(fname, '.');
29  if (p) *p = 0;
30  return fname;
31 }

33 static char *clean4c(char *fname) {
34  char *p;
35 
36  while((p = strpbrk(fname, "-+ `~!@#$%^&*(){}[]|:;'\",<>?/|=.\\"))) *p = '_';
37  return fname;
38 }

40 int main(int argc, char * argv[])
41 {
42 char * opt_progname;
43 char * opt_inputfile;
44 char * opt_outputfile;
45 char * opt_arrayname;
46 int opt_breakblocks;
47 char * opt_static;
48 char * opt_const;
49 FILE * f_input;
50 FILE * f_output;
51 unsigned blocknum;
52 size_t len;
53 size_t i;
54 
55  /* Default values for our parameters */
56  opt_progname = filenameof(argv[0]);
57  opt_inputfile = 0;
58  opt_outputfile = 0;
59  opt_arrayname = 0;
60  opt_breakblocks = 0;
61  opt_static = "";
62  opt_const = "";
63 
64  /* Read the arguments */
65  while(*++argv) {
66  if (argv[0][0] == '-') {
67  while (*++(argv[0])) {
68  switch(argv[0][0]) {
69  case '?': case 'h': goto usage;
70  case 'b': opt_breakblocks = 1; break;
71  case 'c': opt_const = "const "; break;
72  case 's': opt_static = "static "; break;
73  case 'n': opt_arrayname = *++argv; goto nextarg;
74  default:
75  fprintf(stderr, "Unknown flag -%c\n", argv[0][0]);
76  goto usage;
77  }
78  }
79  } else if (!opt_inputfile)
80  opt_inputfile = argv[0];
81  else if (!opt_outputfile)
82  opt_outputfile = argv[0];
83  else {
84  usage:
85  fprintf(stderr, "Usage:\n\t%s -?\n"
86  "\t%s [-bs] [-n name] [inputfile] [outputfile]\n"
87  "\t\t-?\tThis help\n"
88  "\t\t-h\tThis help\n"
89  "\t\t-b\tBreak the arrays for compilers that won't handle large arrays\n"
90  "\t\t-c\tDeclare the arrays as const (useful to ensure they end up in Flash)\n"
91  "\t\t-s\tDeclare the arrays as static\n"
92  "\t\t-n name\tUse \"name\" as the name of the array\n"
93  , opt_progname, opt_progname);
94  return 1;
95  }
96  nextarg: ;
97  }
98 
99  /* Open the input file */
100  if (opt_inputfile) {
101  f_input = fopen(opt_inputfile,
102 #ifdef WIN32
103  "rb");
104 #else
105  "r");
106 #endif
107  if (!f_input) {
108  fprintf(stderr, "Could not open input file '%s'\n", opt_inputfile);
109  goto usage;
110  }
111  } else {
112  f_input = stdin;
113 #ifdef WIN32
114  _setmode(_fileno(stdin), _O_BINARY);
115 #endif
116  }
117 
118  /* Open the output file */
119  if (opt_outputfile) {
120  f_output = fopen(opt_outputfile, "w");
121  if (!f_output) {
122  fprintf(stderr, "Could not open output file '%s'\n", opt_outputfile);
123  goto usage;
124  }
125  } else
126  f_output = stdout;
127 
128  /* Print the comment header */
129  fprintf(f_output, "/**\n * This file was generated ");
130  if (opt_inputfile) fprintf(f_output, "from \"%s\" ", opt_inputfile);
131  fprintf(f_output, "using...\n *\n *\t%s", opt_progname);
132  if (opt_arrayname || opt_static[0] || opt_const[0] || opt_breakblocks) {
133  fprintf(f_output, " -");
134  if (opt_breakblocks) fprintf(f_output, "b");
135  if (opt_const[0]) fprintf(f_output, "c");
136  if (opt_static[0]) fprintf(f_output, "s");
137  if (opt_arrayname) fprintf(f_output, "n %s", opt_arrayname);
138  }
139  if (opt_inputfile) fprintf(f_output, " %s", opt_inputfile);
140  if (opt_outputfile) fprintf(f_output, " %s", opt_outputfile);
141  fprintf(f_output, "\n *\n */\n");
142 
143  /*
144  * Set the array name.
145  * We do this after printing opt_inputfile for the last time as we
146  * modify opt_inputfile in place to generate opt_arrayname.
147  */
148  if (!opt_arrayname) {
149  if (opt_inputfile)
150  opt_arrayname = filenameof(opt_inputfile);
151  if (!opt_arrayname || !opt_arrayname[0])
152  opt_arrayname = "filearray";
153  }
154  opt_arrayname = clean4c(opt_arrayname);
155 
156  /* Read the file processing 1K at a time */
157  blocknum = 0;
158  while((len = fread(buf, 1, sizeof(buf), f_input))) {
159  if (!blocknum++)
160  fprintf(f_output, "%s%sunsigned char %s[] = {", opt_static, opt_const, opt_arrayname);
161  else if (opt_breakblocks)
162  fprintf(f_output, "\n};\n%s%sunsigned char %s_p%u[] = {", opt_static, opt_const, opt_arrayname, blocknum);
163  for(i = 0; i < len; i++) {
164  fprintf(f_output, (i & 0x0F) ? " 0x%02X," : "\n\t0x%02X,", buf[i]);
165  }
166  }
167  fprintf(f_output, "\n};\n");
168 
169  /* Clean up */
170  if (ferror(f_input))
171  fprintf(stderr, "Input file read error\n");
172  if (ferror(f_output))
173  fprintf(stderr, "Output file write error - disk full?\n");
174  if (f_input != stdin)
175  fclose(f_input);
176  if (f_output != stdout)
177  fclose(f_output);
178 
179  return 0;
180 }
