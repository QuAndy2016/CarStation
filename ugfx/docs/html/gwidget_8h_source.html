96 /**
97  * @brief The GWIN Widget structure
98  * @note A widget is a GWIN window that accepts user input.
99  * It also has a number of other properties such as its ability
100  * to redraw itself (a widget maintains drawing state).
101  * @note Do not access the members directly. Treat it as a black-box and use the method functions.
102  *
103  * @{
104  */
105 typedef struct GWidgetObject {
106  GWindowObject g; // @< This is still a GWIN
107  const char * text; // @< The widget text
108  CustomWidgetDrawFunction fnDraw; // @< The current draw function
109  void * fnParam; // @< A parameter for the current draw function
110  const GWidgetStyle * pstyle; // @< The current widget style colors
111 } GWidgetObject;
112 /* @} */

114 /**
115  * A comment/rant on the above structure:
116  * We would really like the GWindowObject member to be anonymous. While this is
117  * allowed under the C11, C99, GNU and various other standards which have been
118  * around forever - compiler support often requires special flags e.g
119  * gcc requires the -fms-extensions flag (no wonder the language and compilers have
120  * not really progressed in 30 years). As portability is a key requirement
121  * we unfortunately won't use this useful feature in case we get a compiler that
122  * won't support it even with special flags.
