165 gdispImageError gdispImageOpen(gdispImage *img) {
166  gdispImageError err;
167 
168  img->bgcolor = White;
169  for(img->fns = ImageHandlers; img->fns < ImageHandlers+sizeof(ImageHandlers)/sizeof(ImageHandlers[0]); img->fns++) {
170  err = img->fns->open(img);
171  if (err != GDISP_IMAGE_ERR_BADFORMAT) {
172  if ((err & GDISP_IMAGE_ERR_UNRECOVERABLE))
173  img->fns = 0;
174  return err;
175  }
176  img->io.fns->seek(&img->io, 0);
177  }
178  img->type = GDISP_IMAGE_TYPE_UNKNOWN;
179  img->flags = 0;
180  img->fns = 0;
181  img->priv = 0;
182  return GDISP_IMAGE_ERR_BADFORMAT;
183 }
