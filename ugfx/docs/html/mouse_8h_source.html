34 typedef struct GEventMouse_t {
35  GEventType type; // The type of this event (GEVENT_MOUSE or GEVENT_TOUCH)
36  uint16_t instance; // The mouse/touch instance
37  coord_t x, y, z; // The position of the mouse.
38  // - For touch devices, Z is the current pressure if supported (otherwise 0)
39  // - For mice, Z is the 3rd dimension if supported (otherwise 0)
40  uint16_t current_buttons; // A bit is set if the button is down.
41  // - For touch only bit 0 is relevant
42  // - For mice the order of the buttons is (from 0 to n) left, right, middle, any other buttons
43  // - Bit 15 being set indicates that an important mouse event has been missed.
44  #define GINPUT_MOUSE_BTN_LEFT 0x0001
45  #define GINPUT_MOUSE_BTN_RIGHT 0x0002
46  #define GINPUT_MOUSE_BTN_MIDDLE 0x0004
47  #define GINPUT_MOUSE_BTN_4 0x0008
48  #define GINPUT_MISSED_MOUSE_EVENT 0x8000
49  #define GINPUT_TOUCH_PRESSED GINPUT_MOUSE_BTN_LEFT
50  uint16_t last_buttons; // The value of current_buttons on the last event
51  enum GMouseMeta_e {
52  GMETA_NONE = 0, // There is no meta event currently happening
53  GMETA_MOUSE_DOWN = 1, // Button 0 has just gone down
54  GMETA_MOUSE_UP = 2, // Button 0 has just gone up
55  GMETA_MOUSE_CLICK = 4, // Button 0 has just gone through a short down - up cycle
56  GMETA_MOUSE_CXTCLICK = 8 // For mice - The right button has just been depressed
57  // For touch - a long press has just occurred
58  } meta;
59  GDisplay * display; // The display this mouse is currently associated with.
60  } GEventMouse;
