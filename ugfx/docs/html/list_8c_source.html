56 static void sendListEvent(GWidgetObject *gw, int item) {
57  GSourceListener* psl;
58  GEvent* pe;
59 
60  // Trigger a GWIN list event
61  psl = 0;
62 
63  while ((psl = geventGetSourceListener(GWIDGET_SOURCE, psl))) {
64  if (!(pe = geventGetEventBuffer(psl)))
65  continue;
66 
67  ple->type = GEVENT_GWIN_LIST;
68  ple->list = (GHandle)gw;
69  ple->item = item;
70 
71  geventSendEvent(psl);
72  }
73 }

75 static void gwinListDefaultDraw(GWidgetObject* gw, void* param) {
76  (void)param;
77 
78  #if GDISP_NEED_CONVEX_POLYGON
79  static const point upArrow[] = { {0, ARROW}, {ARROW, ARROW}, {ARROW/2, 0} };
80  static const point downArrow[] = { {0, 0}, {ARROW, 0}, {ARROW/2, ARROW} };
81  #endif
82 
83  const gfxQueueASyncItem* qi;
84  int i;
85  coord_t x, y, iheight, iwidth;
86  color_t fill;
87  const GColorSet * ps;
88  #if GWIN_NEED_LIST_IMAGES
89  coord_t sy;
90  #endif
91 
92  ps = (gw->g.flags & GWIN_FLG_ENABLED) ? &gw->pstyle->enabled : &gw->pstyle->disabled;
93  iheight = gdispGetFontMetric(gw->g.font, fontHeight) + TEXTGAP;
94  x = 1;
95 
96  // the scroll area
97  if ((gw2obj->cnt > (gw->g.height-2) / iheight) || (gw->g.flags & GLIST_FLG_SCROLLALWAYS)) {
98  iwidth = gw->g.width - (SCROLLWIDTH+3);
99  gdispGFillArea(gw->g.display, gw->g.x+iwidth+2, gw->g.y+1, SCROLLWIDTH, gw->g.height-2, gdispBlendColor(ps->fill, gw->pstyle->background, 128));
100  gdispGDrawLine(gw->g.display, gw->g.x+iwidth+1, gw->g.y+1, gw->g.x+iwidth+1, gw->g.y+gw->g.height-2, ps->edge);
101  #if GDISP_NEED_CONVEX_POLYGON
102  gdispGFillConvexPoly(gw->g.display, gw->g.x+iwidth+((SCROLLWIDTH-ARROW)/2+2), gw->g.y+(ARROW/2+1), upArrow, 3, ps->fill);
103  gdispGFillConvexPoly(gw->g.display, gw->g.x+iwidth+((SCROLLWIDTH-ARROW)/2+2), gw->g.y+gw->g.height-(ARROW+ARROW/2+1), downArrow, 3, ps->fill);
104  #else
105  #warning "GWIN: Lists display better when GDISP_NEED_CONVEX_POLGON is turned on"
106  gdispGFillArea(gw->g.display, gw->g.x+iwidth+((SCROLLWIDTH-ARROW)/2+2), gw->g.y+(ARROW/2+1), ARROW, ARROW, ps->fill);
107  gdispGFillArea(gw->g.display, gw->g.x+iwidth+((SCROLLWIDTH-ARROW)/2+2), gw->g.y+gw->g.height-(ARROW+ARROW/2+1), ARROW, ARROW, ps->fill);
108  #endif
109  } else
110  iwidth = gw->g.width - 2;
111 
112  #if GWIN_NEED_LIST_IMAGES
113  if ((gw->g.flags & GLIST_FLG_HASIMAGES)) {
114  x += iheight;
115  iwidth -= iheight;
116  }
117  #endif
118 
119 
120  // Find the top item
121  for (qi = gfxQueueASyncPeek(&gw2obj->list_head), i = 0; i < gw2obj->top && qi; qi = gfxQueueASyncNext(qi), i++);
122 
123  // Draw until we run out of room or items
124  for (y=1; y+iheight < gw->g.height-1 && qi; qi = gfxQueueASyncNext(qi), y += iheight) {
125  fill = (qi2li->flags & GLIST_FLG_SELECTED) ? ps->fill : gw->pstyle->background;
126  #if GWIN_NEED_LIST_IMAGES
127  if ((gw->g.flags & GLIST_FLG_HASIMAGES)) {
128  // Clear the image area
129  gdispGFillArea(gw->g.display, gw->g.x+1, gw->g.y+y, x-1, iheight, fill);
130  if (qi2li->pimg && gdispImageIsOpen(qi2li->pimg)) {
131  // Calculate which image
132  sy = (qi2li->flags & GLIST_FLG_SELECTED) ? 0 : (iheight-TEXTGAP);
133  if (!(gw->g.flags & GWIN_FLG_ENABLED))
134  sy += 2*(iheight-TEXTGAP);
135  while (sy > qi2li->pimg->height)
136  sy -= iheight-TEXTGAP;
137  // Draw the image
138  gdispImageSetBgColor(qi2li->pimg, fill);
139  gdispGImageDraw(gw->g.display, qi2li->pimg, gw->g.x+1, gw->g.y+y, iheight-TEXTGAP, iheight-TEXTGAP, 0, sy);
140  }
141  }
142  #endif
143  gdispGFillStringBox(gw->g.display, gw->g.x+x, gw->g.y+y, iwidth, iheight, qi2li->text, gw->g.font, ps->text, fill, justifyLeft);
144  }
145 
146  // Fill any remaining item space
147  if (y < gw->g.height-1)
148  gdispGFillArea(gw->g.display, gw->g.x+1, gw->g.y+y, iwidth, gw->g.height-1-y, gw->pstyle->background);
149 
150  // the list frame
151  gdispGDrawBox(gw->g.display, gw->g.x, gw->g.y, gw->g.width, gw->g.height, ps->edge);
152 }

154 #if GINPUT_NEED_MOUSE
155  // a mouse down has occurred over the list area
156  static void MouseDown(GWidgetObject* gw, coord_t x, coord_t y) {
157  const gfxQueueASyncItem* qi;
158  int item, i, pgsz;
159  coord_t iheight;
160  (void) x;
161 
162  iheight = gdispGetFontMetric(gw->g.font, fontHeight) + TEXTGAP;
163  pgsz = (gw->g.height-2)/iheight;
164  if (pgsz < 1) pgsz = 1;
165 
166  // Handle click over the scroll bar
167  if (gw2obj->cnt > pgsz && x >= gw->g.width-(SCROLLWIDTH+2)) {
168  if (y < 2*ARROW) {
169  if (gw2obj->top > 0) {
170  gw2obj->top--;
171  _gwidgetRedraw(&gw->g);
172  }
173  } else if (y >= gw->g.height - 2*ARROW) {
174  if (gw2obj->top < gw2obj->cnt - pgsz) {
175  gw2obj->top++;
176  _gwidgetRedraw(&gw->g);
177  }
178  } else if (y < gw->g.height/2) {
179  if (gw2obj->top > 0) {
180  if (gw2obj->top > pgsz)
181  gw2obj->top -= pgsz;
182  else
183  gw2obj->top = 0;
184  _gwidgetRedraw(&gw->g);
185  }
186  } else {
187  if (gw2obj->top < gw2obj->cnt - pgsz) {
188  if (gw2obj->top < gw2obj->cnt - 2*pgsz)
189  gw2obj->top += pgsz;
190  else
191  gw2obj->top = gw2obj->cnt - pgsz;
192  _gwidgetRedraw(&gw->g);
193  }
194  }
195  return;
196  }
197 
198  // Handle click over the list area
199  item = gw2obj->top + y / iheight;
200 
201  if (item < 0 || item >= gw2obj->cnt)
202  return;
203 
204  for(qi = gfxQueueASyncPeek(&gw2obj->list_head), i = 0; qi; qi = gfxQueueASyncNext(qi), i++) {
205  if ((gw->g.flags & GLIST_FLG_MULTISELECT)) {
206  if (item == i) {
207  qi2li->flags ^= GLIST_FLG_SELECTED;
208  break;
209  }
210  } else {
211  if (item == i)
212  qi2li->flags |= GLIST_FLG_SELECTED;
213  else
214  qi2li->flags &=~ GLIST_FLG_SELECTED;
215  }
216  }
217 
218  _gwidgetRedraw(&gw->g);
219  sendListEvent(gw, item);
220  }
221 #endif

223 #if GINPUT_NEED_TOGGLE
224  // a toggle-on has occurred
225  static void ToggleOn(GWidgetObject *gw, uint16_t role) {
226  const gfxQueueASyncItem * qi;
227  const gfxQueueASyncItem * qix;
228  int i;
229 
230  switch (role) {
231  // select down
232  case 0:
233  for (i = 0, qi = gfxQueueASyncPeek(&gw2obj->list_head); qi; qi = gfxQueueASyncNext(qi), i++) {
234  if ((qi2li->flags & GLIST_FLG_SELECTED)) {
235  qix = gfxQueueASyncNext(qi);
236  if (qix) {
237  qi2li->flags &=~ GLIST_FLG_SELECTED;
238  qix2li->flags |= GLIST_FLG_SELECTED;
239  _gwidgetRedraw(&gw->g);
240  }
241  break;
242  }
243  }
244  break;
245 
246  // select up
247  case 1:
248  qi = gfxQueueASyncPeek(&gw2obj->list_head);
249  qix = 0;
250 
251  for (i = 0; qi; qix = qi, qi = gfxQueueASyncNext(qi), i++) {
252  if ((qi2li->flags & GLIST_FLG_SELECTED))
253  if (qix) {
254  qi2li->flags &=~ GLIST_FLG_SELECTED;
255  qix2li->flags |= GLIST_FLG_SELECTED;
256  _gwidgetRedraw(&gw->g);
257  }
258  break;
259  }
260  }
261  break;
262  }
263  }

354 int gwinListAddItem(GHandle gh, const char* item_name, bool_t useAlloc) {
355  ListItem *newItem;
356 
357  if (useAlloc) {
358  if (!(newItem = (ListItem *)gfxAlloc(sizeof(ListItem)+strlen(item_name)+1)))
359  return -1;
360 
361  strcpy((char *)(newItem+1), item_name);
362  item_name = (const char *)(newItem+1);
363  } else {
364  if (!(newItem = (ListItem *)gfxAlloc(sizeof(ListItem))))
365  return -1;
366  }
367 
368  // the item is not selected when added
369  newItem->flags = 0;
370  newItem->param = 0;
371  newItem->text = item_name;
372  #if GWIN_NEED_LIST_IMAGES
373  newItem->pimg = 0;
374  #endif
375 
376  // select the item if it's the first in the list
377  if (gh2obj->cnt == 0 && !(gh->flags & GLIST_FLG_MULTISELECT))
378  newItem->flags |= GLIST_FLG_SELECTED;
379 
380  // add the new item to the list
381  gfxQueueASyncPut(&gh2obj->list_head, &newItem->q_item);
382 
383  // increment the total amount of entries in the list widget
384  gh2obj->cnt++;
385 
386  _gwidgetRedraw(gh);
387 
388  // return the position in the list (-1 because we start with index 0)
389  return gh2obj->cnt-1;
390 }
