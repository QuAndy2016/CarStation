36 static void _redraw(GHandle gh) {
37  coord_t x, y, w, h, dx, dy;
38  color_t bg;
39  #if GWIN_NEED_IMAGE_ANIMATION
40  delaytime_t delay;
41  #endif
42 
43  // The default display area
44  dx = 0;
45  dy = 0;
46  x = gh->x;
47  y = gh->y;
48  w = gh->width;
49  h = gh->height;
50  bg = gwinGetDefaultBgColor();
51 
52  // If the image isn't open just clear the area
53  if (!gdispImageIsOpen(&widget(gh)->image)) {
54  gdispGFillArea(gh->display, x, y, w, h, bg);
55  return;
56  }
57 
58  // Center horizontally if the area is larger than the image
59  if (widget(gh)->image.width < w) {
60  w = widget(gh)->image.width;
61  dx = (gh->width-w)/2;
62  x += dx;
63  if (dx)
64  gdispGFillArea(gh->display, gh->x, y, dx, h, bg);
65  gdispGFillArea(gh->display, x+w, y, gh->width-dx-w, h, bg);
66  dx = 0;
67  }
68 
69  // Center image horizontally if the area is smaller than the image
70  else if (widget(gh)->image.width > w) {
71  dx = (widget(gh)->image.width - w)/2;
72  }
73 
74  // Center vertically if the area is larger than the image
75  if (widget(gh)->image.height < h) {
76  h = widget(gh)->image.height;
77  dy = (gh->height-h)/2;
78  y += dy;
79  if (dy)
80  gdispGFillArea(gh->display, x, gh->y, w, dy, bg);
81  gdispGFillArea(gh->display, x, y+h, w, gh->height-dy-h, bg);
82  dy = 0;
83  }
84 
85  // Center image vertically if the area is smaller than the image
86  else if (widget(gh)->image.height > h) {
87  dy = (widget(gh)->image.height - h)/2;
88  }
89 
90  // Reset the background color in case it has changed
91  gdispImageSetBgColor(&widget(gh)->image, bg);
92 
93  // Display the image
94  gdispGImageDraw(gh->display, &widget(gh)->image, x, y, w, h, dx, dy);
95 
96  #if GWIN_NEED_IMAGE_ANIMATION
97  // read the delay for the next frame
98  delay = gdispImageNext(&widget((GHandle)gh)->image);
99 
100  // Wait for that delay if required
101  switch(delay) {
102  case TIME_INFINITE:
103  // Everything is done
104  break;
105  case TIME_IMMEDIATE:
106  // We can't allow a continuous loop here as it would lock the system up so we delay for the minimum period
107  delay = 1;
108  // Fall through
109  default:
110  // Start the timer to draw the next frame of the animation
111  gtimerStart(&widget((GHandle)gh)->timer, _timer, (void*)gh, FALSE, delay);
112  break;
113  }
114  #endif
115 }

142 bool_t gwinImageOpenMemory(GHandle gh, const void* memory) {
143  if (gdispImageIsOpen(&widget(gh)->image))
144  gdispImageClose(&widget(gh)->image);
145 
146  if (!gdispImageSetMemoryReader(&widget(gh)->image, memory))
147  return FALSE;
148 
149  if (gdispImageOpen(&widget(gh)->image) != GDISP_IMAGE_ERR_OK)
150  return FALSE;
151 
152  if ((gh->flags & GWIN_FLG_VISIBLE)) {
153  // Setting the clip here shouldn't be necessary if the redraw doesn't overdraw
154  // but we put it in for safety anyway
155  #if GDISP_NEED_CLIP
156  gdispGSetClip(gh->display, gh->x, gh->y, gh->width, gh->height);
157  #endif
158  _redraw(gh);
159  }
160 
161  return TRUE;
162 }
