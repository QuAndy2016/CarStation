106  static inline void _tsDo3PointCalibration(const MousePoint *cross, const MousePoint *points, Calibration *c) {
107  float dx, dx0, dx1, dx2, dy0, dy1, dy2;
108 
109  /* Compute all the required determinants */
110  dx = ((float)(points[0].x - points[2].x)) * ((float)(points[1].y - points[2].y))
111  - ((float)(points[1].x - points[2].x)) * ((float)(points[0].y - points[2].y));
112 
113  dx0 = ((float)(cross[0].x - cross[2].x)) * ((float)(points[1].y - points[2].y))
114  - ((float)(cross[1].x - cross[2].x)) * ((float)(points[0].y - points[2].y));
115 
116  dx1 = ((float)(cross[1].x - cross[2].x)) * ((float)(points[0].x - points[2].x))
117  - ((float)(cross[0].x - cross[2].x)) * ((float)(points[1].x - points[2].x));
118 
119  dx2 = cross[0].x * ((float)points[1].x * (float)points[2].y - (float)points[2].x * (float)points[1].y) -
120  cross[1].x * ((float)points[0].x * (float)points[2].y - (float)points[2].x * (float)points[0].y) +
121  cross[2].x * ((float)points[0].x * (float)points[1].y - (float)points[1].x * (float)points[0].y);
122 
123  dy0 = ((float)(cross[0].y - cross[2].y)) * ((float)(points[1].y - points[2].y))
124  - ((float)(cross[1].y - cross[2].y)) * ((float)(points[0].y - points[2].y));
125 
126  dy1 = ((float)(cross[1].y - cross[2].y)) * ((float)(points[0].x - points[2].x))
127  - ((float)(cross[0].y - cross[2].y)) * ((float)(points[1].x - points[2].x));
128 
129  dy2 = cross[0].y * ((float)points[1].x * (float)points[2].y - (float)points[2].x * (float)points[1].y) -
130  cross[1].y * ((float)points[0].x * (float)points[2].y - (float)points[2].x * (float)points[0].y) +
131  cross[2].y * ((float)points[0].x * (float)points[1].y - (float)points[1].x * (float)points[0].y);
132 
133  /* Now, calculate all the required coefficients */
134  c->ax = dx0 / dx;
135  c->bx = dx1 / dx;
136  c->cx = dx2 / dx;
137 
138  c->ay = dy0 / dx;
139  c->by = dy1 / dx;
140  c->cy = dy2 / dx;
141  }
142 #endif

166 static void get_calibrated_reading(MouseReading *pt) {
167  #if GINPUT_MOUSE_NEED_CALIBRATION || GDISP_NEED_CONTROL
168  coord_t w, h;
169  #endif
170 
171  get_raw_reading(pt);
172 
173  #if GINPUT_MOUSE_NEED_CALIBRATION || GDISP_NEED_CONTROL
174  w = gdispGGetWidth(MouseConfig.display);
175  h = gdispGGetHeight(MouseConfig.display);
176  #endif
177 
178  #if GINPUT_MOUSE_NEED_CALIBRATION
179  _tsTransform(pt, &MouseConfig.caldata);
180  #endif
181 
182  #if GDISP_NEED_CONTROL
183  switch(gdispGGetOrientation(MouseConfig.display)) {
184  case GDISP_ROTATE_0:
185  break;
186  case GDISP_ROTATE_90:
187  {
188  coord_t t = pt->x;
189  pt->x = w - 1 - pt->y;
190  pt->y = t;
191  }
192  break;
193  case GDISP_ROTATE_180:
194  pt->x = w - 1 - pt->x;
195  pt->y = h - 1 - pt->y;
196  break;
197  case GDISP_ROTATE_270:
198  {
199  coord_t t = pt->y;
200  pt->y = h - 1 - pt->x;
201  pt->x = t;
202  }
203  break;
204  default:
205  break;
206  }
207  #endif
208 
209  #if GINPUT_MOUSE_NEED_CALIBRATION
210  if (!(MouseConfig.flags & FLG_CAL_RAW)) {
211  if (pt->x < 0) pt->x = 0;
212  else if (pt->x >= w) pt->x = w-1;
213  if (pt->y < 0) pt->y = 0;
214  else if (pt->y >= h) pt->y = h-1;
215  }
216  #endif
217 }

219 static void MousePoll(void *param) {
220  (void) param;
221  GSourceListener *psl;
222  GEventMouse *pe;
223  unsigned meta;
224  uint16_t tbtns;
225  uint32_t cdiff;
226  uint32_t mdiff;
227 
228  // Save the last mouse state
229  MouseConfig.last_buttons = MouseConfig.t.buttons;
230 
231  // Get the new mouse reading
232  get_calibrated_reading(&MouseConfig.t);
233 
234  // Calculate out new event meta value and handle CLICK and CXTCLICK
235  meta = GMETA_NONE;
236 
237  // Calculate the position difference from our movement reference (update the reference if out of range)
238  mdiff = (MouseConfig.t.x - MouseConfig.movepos.x) * (MouseConfig.t.x - MouseConfig.movepos.x) +
239  (MouseConfig.t.y - MouseConfig.movepos.y) * (MouseConfig.t.y - MouseConfig.movepos.y);
240  if (mdiff > GINPUT_MOUSE_MAX_MOVE_JITTER * GINPUT_MOUSE_MAX_MOVE_JITTER) {
241  MouseConfig.movepos.x = MouseConfig.t.x;
242  MouseConfig.movepos.y = MouseConfig.t.y;
243  }
244 
245  // Check if the click has moved outside the click area and if so cancel the click
246  if ((MouseConfig.flags & FLG_CLICK_TIMER)) {
247  cdiff = (MouseConfig.t.x - MouseConfig.clickpos.x) * (MouseConfig.t.x - MouseConfig.clickpos.x) +
248  (MouseConfig.t.y - MouseConfig.clickpos.y) * (MouseConfig.t.y - MouseConfig.clickpos.y);
249  if (cdiff > GINPUT_MOUSE_MAX_CLICK_JITTER * GINPUT_MOUSE_MAX_CLICK_JITTER)
250  MouseConfig.flags &= ~FLG_CLICK_TIMER;
251  }
252 
253  // Mouse down
254  tbtns = MouseConfig.t.buttons & ~MouseConfig.last_buttons;
255  if ((tbtns & GINPUT_MOUSE_BTN_LEFT))
256  meta |= GMETA_MOUSE_DOWN;
257  if ((tbtns & (GINPUT_MOUSE_BTN_LEFT|GINPUT_MOUSE_BTN_RIGHT))) {
258  MouseConfig.clickpos.x = MouseConfig.t.x;
259  MouseConfig.clickpos.y = MouseConfig.t.y;
260  MouseConfig.clicktime = gfxSystemTicks();
261  MouseConfig.flags |= FLG_CLICK_TIMER;
262  }
263 
264  // Mouse up
265  tbtns = ~MouseConfig.t.buttons & MouseConfig.last_buttons;
266  if ((tbtns & GINPUT_MOUSE_BTN_LEFT))
267  meta |= GMETA_MOUSE_UP;
268  if ((tbtns & (GINPUT_MOUSE_BTN_LEFT|GINPUT_MOUSE_BTN_RIGHT))) {
269  if ((MouseConfig.flags & FLG_CLICK_TIMER)) {
270  if ((tbtns & GINPUT_MOUSE_BTN_LEFT)
271  #if GINPUT_MOUSE_CLICK_TIME != TIME_INFINITE
272  && gfxSystemTicks() - MouseConfig.clicktime < gfxMillisecondsToTicks(GINPUT_MOUSE_CLICK_TIME)
273  #endif
274  )
275  meta |= GMETA_MOUSE_CLICK;
276  else
277  meta |= GMETA_MOUSE_CXTCLICK;
278  MouseConfig.flags &= ~FLG_CLICK_TIMER;
279  }
280  }
281 
282  // Send the event to the listeners that are interested.
283  psl = 0;
284  while ((psl = geventGetSourceListener((GSourceHandle)(&MouseConfig), psl))) {
285  if (!(pe = (GEventMouse *)geventGetEventBuffer(psl))) {
286  // This listener is missing - save the meta events that have happened
287  psl->srcflags |= meta;
288  continue;
289  }
290 
291  // If we haven't really moved (and there are no meta events) don't bother sending the event
292  if (mdiff <= GINPUT_MOUSE_MAX_MOVE_JITTER * GINPUT_MOUSE_MAX_MOVE_JITTER && !psl->srcflags && !meta && !(psl->listenflags & GLISTEN_MOUSENOFILTER))
293  continue;
294 
295  // Send the event if we are listening for it
296  if (((MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT) && (psl->listenflags & GLISTEN_MOUSEDOWNMOVES))
297  || (!(MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT) && (psl->listenflags & GLISTEN_MOUSEUPMOVES))
298  || (meta && (psl->listenflags & GLISTEN_MOUSEMETA))) {
299  pe->type = GINPUT_MOUSE_EVENT_TYPE;
300  pe->instance = 0;
301  pe->x = MouseConfig.t.x;
302  pe->y = MouseConfig.t.y;
303  pe->z = MouseConfig.t.z;
304  pe->current_buttons = MouseConfig.t.buttons;
305  pe->last_buttons = MouseConfig.last_buttons;
306  pe->meta = meta;
307  if (psl->srcflags) {
308  pe->current_buttons |= GINPUT_MISSED_MOUSE_EVENT;
309  pe->meta |= psl->srcflags;
310  psl->srcflags = 0;
311  }
312  pe->display = MouseConfig.display;
313  geventSendEvent(psl);
314  }
315  }
316 }

318 GSourceHandle ginputGetMouse(uint16_t instance) {
319  #if GINPUT_MOUSE_NEED_CALIBRATION
320  Calibration *pc;
321  #endif
322 
323  // We only support a single mouse instance currently
324  // Instance 9999 is the same as instance 0 except that it installs
325  // a special "raw" calibration if there isn't one we can load.
326  if (instance && instance != 9999)
327  return 0;
328 
329  // Make sure we have a valid mouse display
330  if (!MouseConfig.display)
331  MouseConfig.display = GDISP;
332 
333  // Do we need to initialise the mouse subsystem?
334  if (!(MouseConfig.flags & FLG_INIT_DONE)) {
335  ginput_lld_mouse_init();
336 
337  #if GINPUT_MOUSE_NEED_CALIBRATION
338  #if GINPUT_MOUSE_LLD_CALIBRATION_LOADSAVE
339  if (!MouseConfig.fnloadcal) {
340  MouseConfig.fnloadcal = ginput_lld_mouse_calibration_load;
341  MouseConfig.flags &= ~FLG_CAL_FREE;
342  }
343  if (!MouseConfig.fnsavecal)
344  MouseConfig.fnsavecal = ginput_lld_mouse_calibration_save;
345  #endif
346  if (MouseConfig.fnloadcal && (pc = (Calibration *)MouseConfig.fnloadcal(instance))) {
347  MouseConfig.caldata = pc[0];
348  MouseConfig.flags |= (FLG_CAL_OK|FLG_CAL_SAVED);
349  if ((MouseConfig.flags & FLG_CAL_FREE))
350  gfxFree((void *)pc);
351  } else if (instance == 9999) {
352  MouseConfig.caldata.ax = 1;
353  MouseConfig.caldata.bx = 0;
354  MouseConfig.caldata.cx = 0;
355  MouseConfig.caldata.ay = 0;
356  MouseConfig.caldata.by = 1;
357  MouseConfig.caldata.cy = 0;
358  MouseConfig.flags |= (FLG_CAL_OK|FLG_CAL_SAVED|FLG_CAL_RAW);
359  } else
360  ginputCalibrateMouse(instance);
361  #endif
362 
363  // Get the first reading
364  MouseConfig.last_buttons = 0;
365  get_calibrated_reading(&MouseConfig.t);
366 
367  // Mark init as done and start the Poll timer
368  MouseConfig.flags |= FLG_INIT_DONE;
369  gtimerStart(&MouseTimer, MousePoll, 0, TRUE, GINPUT_MOUSE_POLL_PERIOD);
370  }
371 
372  // Return our structure as the handle
373  return (GSourceHandle)&MouseConfig;
374 }

390 bool_t ginputGetMouseStatus(uint16_t instance, GEventMouse *pe) {
391  // Win32 threads don't seem to recognise priority and/or pre-emption
392  // so we add a sleep here to prevent 100% polled applications from locking up.
393  gfxSleepMilliseconds(1);
394 
395  if (instance || (MouseConfig.flags & (FLG_INIT_DONE|FLG_IN_CAL)) != FLG_INIT_DONE)
396  return FALSE;
397 
398  pe->type = GINPUT_MOUSE_EVENT_TYPE;
399  pe->instance = instance;
400  pe->x = MouseConfig.t.x;
401  pe->y = MouseConfig.t.y;
402  pe->z = MouseConfig.t.z;
403  pe->current_buttons = MouseConfig.t.buttons;
404  pe->last_buttons = MouseConfig.last_buttons;
405  if (pe->current_buttons & ~pe->last_buttons & GINPUT_MOUSE_BTN_LEFT)
406  pe->meta = GMETA_MOUSE_DOWN;
407  else if (~pe->current_buttons & pe->last_buttons & GINPUT_MOUSE_BTN_LEFT)
408  pe->meta = GMETA_MOUSE_UP;
409  else
410  pe->meta = GMETA_NONE;
411  return TRUE;
412 }

414 bool_t ginputCalibrateMouse(uint16_t instance) {
415  #if !GINPUT_MOUSE_NEED_CALIBRATION
416  (void) instance;
417 
418  return FALSE;
419  #else
420 
421  const coord_t height = gdispGGetHeight(MouseConfig.display);
422  const coord_t width = gdispGGetWidth(MouseConfig.display);
423  const MousePoint cross[] = {{(width / 4), (height / 4)},
424  {(width - (width / 4)) , (height / 4)},
425  {(width - (width / 4)) , (height - (height / 4))},
426  {(width / 2), (height / 2)}}; /* Check point */
427  MousePoint points[GINPUT_MOUSE_CALIBRATION_POINTS];
428  const MousePoint *pc;
429  MousePoint *pt;
430  int32_t px, py;
431  unsigned i, j;
432  font_t font1, font2;
433  #if GINPUT_MOUSE_MAX_CALIBRATION_ERROR >= 0
434  unsigned err;
435  #endif
436 
437  if (instance || (MouseConfig.flags & FLG_IN_CAL))
438  return FALSE;
439 
440  font1 = gdispOpenFont(GINPUT_MOUSE_CALIBRATION_FONT);
441  font2 = gdispOpenFont(GINPUT_MOUSE_CALIBRATION_FONT2);
442 
443  MouseConfig.flags |= FLG_IN_CAL;
444  gtimerStop(&MouseTimer);
445  MouseConfig.flags &= ~(FLG_CAL_OK|FLG_CAL_SAVED|FLG_CAL_RAW);
446 
447  #if GDISP_NEED_CONTROL
448  gdispGSetOrientation(MouseConfig.display, GDISP_ROTATE_0);
449  #endif
450 
451  #if GDISP_NEED_CLIP
452  gdispGSetClip(MouseConfig.display, 0, 0, width, height);
453  #endif
454 
455  #if GINPUT_MOUSE_MAX_CALIBRATION_ERROR >= 0
456  while(1) {
457  #endif
458  gdispGClear(MouseConfig.display, Blue);
459 
460  gdispGFillStringBox(MouseConfig.display, 0, 5, width, 30, GINPUT_MOUSE_CALIBRATION_TEXT, font1, White, Blue, justifyCenter);
461 
462  for(i = 0, pt = points, pc = cross; i < GINPUT_MOUSE_CALIBRATION_POINTS; i++, pt++, pc++) {
463  _tsDrawCross(pc);
464 
465  do {
466 
467  /* Wait for the mouse to be pressed */
468  while(get_raw_reading(&MouseConfig.t), !(MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT))
469  gfxSleepMilliseconds(20);
470 
471  /* Average all the samples while the mouse is down */
472  for(px = py = 0, j = 0;
473  gfxSleepMilliseconds(20), /* Settling time between readings */
474  get_raw_reading(&MouseConfig.t),
475  (MouseConfig.t.buttons & GINPUT_MOUSE_BTN_LEFT);
476  j++) {
477  px += MouseConfig.t.x;
478  py += MouseConfig.t.y;
479  }
480 
481  } while(!j);
482 
483  pt->x = px / j;
484  pt->y = py / j;
485 
486  _tsClearCross(pc);
487 
488  if (i >= 1 && pt->x == (pt-1)->x && pt->y == (pt-1)->y) {
489  gdispGFillStringBox(MouseConfig.display, 0, 35, width, 40, GINPUT_MOUSE_CALIBRATION_SAME_TEXT, font2, Red, Yellow, justifyCenter);
490  gfxSleepMilliseconds(5000);
491  gdispGFillArea(MouseConfig.display, 0, 35, width, 40, Blue);
492  }
493 
494  }
495 
496  /* Apply 3 point calibration algorithm */
497  _tsDo3PointCalibration(cross, points, &MouseConfig.caldata);
498 
499  /* Verification of correctness of calibration (optional) :
500  * See if the 4th point (Middle of the screen) coincides with the calibrated
501  * result. If point is within +/- Squareroot(ERROR) pixel margin, then successful calibration
502  * Else, start from the beginning.
503  */
504  #if GINPUT_MOUSE_MAX_CALIBRATION_ERROR >= 0
505  /* Transform the co-ordinates */
506  MouseConfig.t.x = points[3].x;
507  MouseConfig.t.y = points[3].y;
508  _tsTransform(&MouseConfig.t, &MouseConfig.caldata);
509 
510  /* Calculate the delta */
511  err = (MouseConfig.t.x - cross[3].x) * (MouseConfig.t.x - cross[3].x) +
512  (MouseConfig.t.y - cross[3].y) * (MouseConfig.t.y - cross[3].y);
513 
514  if (err <= GINPUT_MOUSE_MAX_CALIBRATION_ERROR * GINPUT_MOUSE_MAX_CALIBRATION_ERROR)
515  break;
516 
517  gdispGFillStringBox(MouseConfig.display, 0, 35, width, 40, GINPUT_MOUSE_CALIBRATION_ERROR_TEXT, font2, Red, Yellow, justifyCenter);
518  gfxSleepMilliseconds(5000);
519  }
520  #endif
521 
522  // Restart everything
523  gdispCloseFont(font1);
524  gdispCloseFont(font2);
525  MouseConfig.flags |= FLG_CAL_OK;
526  MouseConfig.last_buttons = 0;
527  get_calibrated_reading(&MouseConfig.t);
528  MouseConfig.flags &= ~FLG_IN_CAL;
529  if ((MouseConfig.flags & FLG_INIT_DONE))
530  gtimerStart(&MouseTimer, MousePoll, 0, TRUE, GINPUT_MOUSE_POLL_PERIOD);
531 
532  // Save the calibration data (if possible)
533  if (MouseConfig.fnsavecal) {
534  MouseConfig.fnsavecal(instance, (const uint8_t *)&MouseConfig.caldata, sizeof(MouseConfig.caldata));
535  MouseConfig.flags |= FLG_CAL_SAVED;
536  }
537 
538  // Clear the screen using the GWIN default background color
539  #if GFX_USE_GWIN
