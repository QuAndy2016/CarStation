89 void gwinPutChar(GHandle gh, char c) {
90  #define gcw ((GConsoleObject *)gh)
91  uint8_t width, fy, fp;
92 
93  if (gh->vmt != &consoleVMT || !gh->font)
94  return;
95 
96  fy = gdispGetFontMetric(gh->font, fontHeight);
97  fp = gdispGetFontMetric(gh->font, fontCharPadding);
98 
99  #if GDISP_NEED_CLIP
100  gdispGSetClip(gh->display, gh->x, gh->y, gh->width, gh->height);
101  #endif
102 
103  if (c == '\n') {
104  gcw->cx = 0;
105  gcw->cy += fy;
106  // We use lazy scrolling here and only scroll when the next char arrives
107  } else if (c == '\r') {
108  // gcw->cx = 0;
109  } else {
110  width = gdispGetCharWidth(c, gh->font) + fp;
111  if (gcw->cx + width >= gh->width) {
112  gcw->cx = 0;
113  gcw->cy += fy;
114  }
115 
116  if (gcw->cy + fy > gh->height) {
117 #if GDISP_NEED_SCROLL
118  /* scroll the console */
119  gdispGVerticalScroll(gh->display, gh->x, gh->y, gh->width, gh->height, fy, gh->bgcolor);
120  /* reset the cursor to the start of the last line */
121  gcw->cx = 0;
122  gcw->cy = (((coord_t)(gh->height/fy))-1)*fy;
123 #else
124  /* clear the console */
125  gdispGFillArea(gh->display, gh->x, gh->y, gh->width, gh->height, gh->bgcolor);
126  /* reset the cursor to the top of the window */
127  gcw->cx = 0;
128  gcw->cy = 0;
129 #endif
130  }
131 
132 #if GWIN_CONSOLE_USE_CLEAR_LINES
133  /* clear to the end of the line */
134  if (gcw->cx == 0)
135  gdispGFillArea(gh->display, gh->x, gh->y + gcw->cy, gh->width, fy, gh->bgcolor);
136 #endif
137 #if GWIN_CONSOLE_USE_FILLED_CHARS
138  gdispGFillChar(gh->display, gh->x + gcw->cx, gh->y + gcw->cy, c, gh->font, gh->color, gh->bgcolor);
139 #else
140  gdispGDrawChar(gh->display, gh->x + gcw->cx, gh->y + gcw->cy, c, gh->font, gh->color);
141 #endif
142 
143  /* update cursor */
144  gcw->cx += width;
145  }
146  #undef gcw
147 }

164 static char *ltoa_wd(char *p, long num, unsigned radix, long divisor) {
165  int i;
166  char *q;
167 
168  if (!divisor) divisor = num;
169 
170  q = p + MAX_FILLER;
171  do {
172  i = (int)(num % radix);
173  i += '0';
174  if (i > '9')
175  i += 'A' - '0' - 10;
176  *--q = i;
177  num /= radix;
178  } while ((divisor /= radix) != 0);
179 
180  i = (int)(p + MAX_FILLER - q);
181  do {
182  *p++ = *q++;
183  } while (--i);
184 
185  return p;
186 }

201 void gwinPrintf(GHandle gh, const char *fmt, ...) {
202  va_list ap;
203  char *p, *s, c, filler;
204  int i, precision, width;
205  bool_t is_long, left_align;
206  long l;
207  #if GWIN_CONSOLE_USE_FLOAT
208  float f;
209  char tmpbuf[2*MAX_FILLER + 1];
210  #else
211  char tmpbuf[MAX_FILLER + 1];
212  #endif
213 
214  if (gh->vmt != &consoleVMT || !gh->font)
215  return;
216 
217  va_start(ap, fmt);
218  while (TRUE) {
219  c = *fmt++;
220  if (c == 0) {
221  va_end(ap);
222  return;
223  }
224  if (c != '%') {
225  gwinPutChar(gh, c);
226  continue;
227  }
228 
229  p = tmpbuf;
230  s = tmpbuf;
231  left_align = FALSE;
232  if (*fmt == '-') {
233  fmt++;
234  left_align = TRUE;
235  }
236  filler = ' ';
237  if (*fmt == '.') {
238  fmt++;
239  filler = '0';
240  }
241  width = 0;
242 
243  while (TRUE) {
244  c = *fmt++;
245  if (c >= '0' && c <= '9')
246  c -= '0';
247  else if (c == '*')
248  c = va_arg(ap, int);
249  else
250  break;
251  width = width * 10 + c;
252  }
253  precision = 0;
254  if (c == '.') {
255  while (TRUE) {
256  c = *fmt++;
257  if (c >= '0' && c <= '9')
258  c -= '0';
259  else if (c == '*')
260  c = va_arg(ap, int);
261  else
262  break;
263  precision = precision * 10 + c;
264  }
265  }
266  /* Long modifier.*/
267  if (c == 'l' || c == 'L') {
268  is_long = TRUE;
269  if (*fmt)
270  c = *fmt++;
271  }
272  else
273  is_long = (c >= 'A') && (c <= 'Z');
274 
275  /* Command decoding.*/
276  switch (c) {
277  case 'c':
278  filler = ' ';
279  *p++ = va_arg(ap, int);
280  break;
281  case 's':
282  filler = ' ';
283  if ((s = va_arg(ap, char *)) == 0)
284  s = "(null)";
285  if (precision == 0)
286  precision = 32767;
287  for (p = s; *p && (--precision >= 0); p++);
288  break;
289  case 'D':
290  case 'd':
291  if (is_long)
292  l = va_arg(ap, long);
293  else
294  l = va_arg(ap, int);
295  if (l < 0) {
296  *p++ = '-';
297  l = -l;
298  }
299  p = ltoa_wd(p, l, 10, 0);
300  break;
301  #if GWIN_CONSOLE_USE_FLOAT
302  case 'f':
303  f = (float) va_arg(ap, double);
304  if (f < 0) {
305  *p++ = '-';
306  f = -f;
307  }
308  p = ftoa(p, f);
309  break;
310  #endif
311  case 'X':
312  case 'x':
313  c = 16;
314  goto unsigned_common;
315  case 'U':
316  case 'u':
317  c = 10;
318  goto unsigned_common;
319  case 'O':
320  case 'o':
321  c = 8;
322  unsigned_common:
323  if (is_long)
324  l = va_arg(ap, long);
325  else
326  l = va_arg(ap, int);
327  p = ltoa_wd(p, l, c, 0);
328  break;
329  default:
330  *p++ = c;
331  break;
332  }
333 
334  i = (int)(p - s);
335  if ((width -= i) < 0)
336  width = 0;
337  if (left_align == FALSE)
338  width = -width;
339  if (width < 0) {
340  if (*s == '-' && filler == '0') {
341  gwinPutChar(gh, *s++);
342  i--;
343  }
344  do {
345  gwinPutChar(gh, filler);
346  } while (++width != 0);
347  }
348  while (--i >= 0)
349  gwinPutChar(gh, *s++);
350  while (width) {
351  gwinPutChar(gh, filler);
352  width--;
353  }
354  }
355 }
