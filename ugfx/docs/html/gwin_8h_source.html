30 /**
31  * @brief A window object structure
32  * @note Do not access the members directly. Treat it as a black-box and use the method functions.
33  * @{
34  */
35 typedef struct GWindowObject {
36  #if GWIN_NEED_WINDOWMANAGER
37  // This MUST be the first member of the struct
38  gfxQueueASyncItem wmq; // @< The next window (for the window manager)
39  #endif
40  const struct gwinVMT *vmt; // @< The VMT for this GWIN
41  GDisplay * display; // @< The display this window is on.
42  coord_t x, y; // @< Screen relative position
43  coord_t width, height; // @< Dimensions of this window
44  color_t color, bgcolor; // @< The current drawing colors
45  uint16_t flags; // @< Window flags (the meaning is private to the GWIN class)
46  #if GDISP_NEED_TEXT
47  font_t font; // @< The current font
48  #endif
49 } GWindowObject, * GHandle;
50 /* @} */
