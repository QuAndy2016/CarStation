25 /**
26  * Match a pattern against the font name.
27  */
28 static bool_t matchfont(const char *pattern, const char *name) {
29  while(1) {
30  switch (pattern[0]) {
31  case '*':
32  if (name[0] == 0)
33  return pattern[1] == 0;
34  if (pattern[1] == name[0])
35  pattern++;
36  else
37  name++;
38  break;
39  case 0:
40  return name[0] == 0;
41  default:
42  if (name[0] != pattern[0])
43  return FALSE;
44  pattern++;
45  name++;
46  break;
47  }
48  }
49 }

51 font_t gdispOpenFont(const char *name) {

55  // Try the long names first
56  for(fp = mf_get_font_list(); fp; fp = fp->next) {
57  if (matchfont(name, fp->font->full_name))
58  return fp->font;

60 
61  // Try the short names if no long names match

63  if (matchfont(name, fp->font->short_name))
