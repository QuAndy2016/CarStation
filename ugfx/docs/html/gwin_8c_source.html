41  static void _gwm_vis(GHandle gh) {
42  if (gh->vmt->Redraw) {
43  #if GDISP_NEED_CLIP
44  gdispSetClip(gh->x, gh->y, gh->width, gh->height);
45  #endif
46  gh->vmt->Redraw(gh);
47  } else
48  gwinClear(gh);

50  static void _gwm_redim(GHandle gh, coord_t x, coord_t y, coord_t width, coord_t height) {
51  gh->x = x; gh->y = y;
52  gh->width = width; gh->height = height;
53  if (gh->x < 0) { gh->width += gh->x; gh->x = 0; }
54  if (gh->y < 0) { gh->height += gh->y; gh->y = 0; }
55  if (gh->x > gdispGetWidth()-MIN_WIN_WIDTH) gh->x = gdispGetWidth()-MIN_WIN_WIDTH;
56  if (gh->y > gdispGetHeight()-MIN_WIN_HEIGHT) gh->y = gdispGetHeight()-MIN_WIN_HEIGHT;
57  if (gh->width < MIN_WIN_WIDTH) { gh->width = MIN_WIN_WIDTH; }
58  if (gh->height < MIN_WIN_HEIGHT) { gh->height = MIN_WIN_HEIGHT; }
59  if (gh->x+gh->width > gdispGetWidth()) gh->width = gdispGetWidth() - gh->x;
60  if (gh->y+gh->height > gdispGetHeight()) gh->height = gdispGetHeight() - gh->y;
61 
62  // Redraw the window
63  if ((gh->flags & GWIN_FLG_VISIBLE)) {
64  if (gh->vmt->Redraw) {
65  #if GDISP_NEED_CLIP
66  gdispSetClip(gh->x, gh->y, gh->width, gh->height);
67  #endif
68  gh->vmt->Redraw(gh);
69  }
70  }
71  }
72 #endif

91 // Internal routine for use by GWIN components only
92 // Initialise a window creating it dynamically if required.
93 GHandle _gwindowCreate(GDisplay *g, GWindowObject *pgw, const GWindowInit *pInit, const gwinVMT *vmt, uint16_t flags) {
94  // Allocate the structure if necessary
95  if (!pgw) {
96  if (!(pgw = (GWindowObject *)gfxAlloc(vmt->size)))
97  return 0;
98  pgw->flags = flags|GWIN_FLG_DYNAMIC;
99  } else
100  pgw->flags = flags;
101 
102  // Initialise all basic fields
103  pgw->display = g;
104  pgw->vmt = vmt;
105  pgw->color = defaultFgColor;
106  pgw->bgcolor = defaultBgColor;
107  #if GDISP_NEED_TEXT
108  pgw->font = defaultFont;
109  #endif
110 
111 #if GWIN_NEED_WINDOWMANAGER
112  if (!_GWINwm->vmt->Add(pgw, pInit)) {
113  if ((pgw->flags & GWIN_FLG_DYNAMIC))
114  gfxFree(pgw);
115  return 0;
116  }
117 #else
118  _gwm_redim(pgw, pInit->x, pInit->y, pInit->width, pInit->height);
119 #endif
120 
121  return (GHandle)pgw;
122 }

165 void gwinDestroy(GHandle gh) {

169  #endif

174 
175  // Clean up the structure

187 void gwinSetVisible(GHandle gh, bool_t visible) {
188  if (visible) {
189  if (!(gh->flags & GWIN_FLG_VISIBLE)) {
190  gh->flags |= GWIN_FLG_VISIBLE;
191  #if GWIN_NEED_WINDOWMANAGER
192  _GWINwm->vmt->Visible(gh);
193  #else
194  _gwm_vis(gh);
195  #endif
196  }
197  } else {
198  if ((gh->flags & GWIN_FLG_VISIBLE)) {
199  gh->flags &= ~GWIN_FLG_VISIBLE;
200  #if GWIN_NEED_WINDOWMANAGER
201  _GWINwm->vmt->Visible(gh);
202  #endif
203  }
204  }
205 }

211 void gwinSetEnabled(GHandle gh, bool_t enabled) {
212  if (enabled) {
213  if (!(gh->flags & GWIN_FLG_ENABLED)) {
214  gh->flags |= GWIN_FLG_ENABLED;
215  if ((gh->flags & GWIN_FLG_VISIBLE) && gh->vmt->Redraw) {
216  #if GDISP_NEED_CLIP
217  gdispGSetClip(gh->display, gh->x, gh->y, gh->width, gh->height);
218  #endif
219  gh->vmt->Redraw(gh);
220  }
221  }
222  } else {
223  if ((gh->flags & GWIN_FLG_ENABLED)) {
224  gh->flags &= ~GWIN_FLG_ENABLED;
225  if ((gh->flags & GWIN_FLG_VISIBLE) && gh->vmt->Redraw) {
226  #if GDISP_NEED_CLIP
227  gdispGSetClip(gh->display, gh->x, gh->y, gh->width, gh->height);
