31 /* Loop through the assignment table deleting this listener/source pair. */
32 /* Null is treated as a wildcard. */
33 static void deleteAssignments(GListener *pl, GSourceHandle gsh) {
34  GSourceListener *psl;
35 
36  for(psl = Assignments; psl < Assignments+GEVENT_MAX_SOURCE_LISTENERS; psl++) {
37  if ((!pl || psl->pListener == pl) && (!gsh || psl->pSource == gsh)) {
38  if (gfxSemCounter(&psl->pListener->waitqueue) < 0) {
39  gfxSemWait(&psl->pListener->eventlock, TIME_INFINITE); // Obtain the buffer lock
40  psl->pListener->event.type = GEVENT_EXIT; // Set up the EXIT event
41  gfxSemSignal(&psl->pListener->waitqueue); // Wake up the listener
42  gfxSemSignal(&psl->pListener->eventlock); // Release the buffer lock
43  }
44  psl->pListener = 0;
45  }
46  }
47 }

60 bool_t geventAttachSource(GListener *pl, GSourceHandle gsh, unsigned flags) {
61  GSourceListener *psl, *pslfree;
62 
63  // Safety first
64  if (!pl || !gsh) {
65  GEVENT_ASSERT(FALSE);
66  return FALSE;
67  }
68 
69  gfxMutexEnter(&geventMutex);
70 
71  // Check if this pair is already in the table (scan for a free slot at the same time)
72  pslfree = 0;
73  for(psl = Assignments; psl < Assignments+GEVENT_MAX_SOURCE_LISTENERS; psl++) {
74 
75  if (pl == psl->pListener && gsh == psl->pSource) {
76  // Just update the flags
77  gfxSemWait(&pl->eventlock, TIME_INFINITE); // Safety first - just in case a source is using it
78  psl->listenflags = flags;
79  gfxSemSignal(&pl->eventlock); // Release this lock
80  gfxMutexExit(&geventMutex);
81  return TRUE;
82  }
83  if (!pslfree && !psl->pListener)
84  pslfree = psl;
85  }
86 
87  // A free slot was found - allocate it
88  if (pslfree) {
89  pslfree->pListener = pl;
90  pslfree->pSource = gsh;
91  pslfree->listenflags = flags;
92  pslfree->srcflags = 0;
93  }
94  gfxMutexExit(&geventMutex);
95  GEVENT_ASSERT(pslfree != 0);
96  return pslfree != 0;
97 }

99 void geventDetachSource(GListener *pl, GSourceHandle gsh) {
100  if (pl) {
101  gfxMutexEnter(&geventMutex);
102  deleteAssignments(pl, gsh);
103  if (!gsh && gfxSemCounter(&pl->waitqueue) < 0) {
104  gfxSemWait(&pl->eventlock, TIME_INFINITE); // Obtain the buffer lock

106  gfxSemSignal(&pl->waitqueue); // Wake up the listener
107  gfxSemSignal(&pl->eventlock); // Release the buffer lock
108  }

134 GSourceListener *geventGetSourceListener(GSourceHandle gsh, GSourceListener *lastlr) {
135  GSourceListener *psl;
136 
137  // Safety first
138  if (!gsh)
139  return 0;
140 
141  gfxMutexEnter(&geventMutex);
142 
143  // Unlock the last listener event buffer
144  if (lastlr)
145  gfxSemSignal(&lastlr->pListener->eventlock);
146 
147  // Loop through the table looking for attachments to this source
148  for(psl = lastlr ? (lastlr+1) : Assignments; psl < Assignments+GEVENT_MAX_SOURCE_LISTENERS; psl++) {
149  if (gsh == psl->pSource) {
150  gfxSemWait(&psl->pListener->eventlock, TIME_INFINITE); // Obtain a lock on the listener event buffer
151  gfxMutexExit(&geventMutex);
152  return psl;
153  }
154  }
155  gfxMutexExit(&geventMutex);
156  return 0;
157 }

164 void geventSendEvent(GSourceListener *psl) {
165  gfxMutexEnter(&geventMutex);
166  if (psl->pListener->callback) { // This test needs to be taken inside the mutex
167  gfxMutexExit(&geventMutex);
168  // We already know we have the event lock
169  psl->pListener->callback(psl->pListener->param, &psl->pListener->event);
170 
171  } else {
172  // Wake up the listener
173  if (gfxSemCounter(&psl->pListener->waitqueue) <= 0)
174  gfxSemSignal(&psl->pListener->waitqueue);
175  gfxMutexExit(&geventMutex);
176  }
177 }
